CREATE TABLE IF NOT EXISTS USUARIO(
ID_USUARIO SERIAL,
USUARIO VARCHAR (100),
CONTRASEÑA VARCHAR(100),
ULTIMO_INGRESO DATE,
ESTADO INT,
ID_EMPLEADO INT,
ID_ROL INT,
PRIMARY KEY (USUARIO,CONTRASEÑA),
FOREIGN KEY (ID_EMPLEADO) REFERENCES EMPLEADO(ID_EMPLEADO),
FOREIGN KEY (ID_ROL) REFERENCES ROL(ID_ROL)
);
